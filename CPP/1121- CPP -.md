template<typename T>
void sortAsc(T* data, const int& size) {
	for (int i = 0; i < size;  i++) {
		for (int j = i + 1; j < size; j++) {
			if (data[i] > data[j]) {
				T temp = data[j];
				data[j] = data[i];
				data[i] = temp;
			}
		}
	}
	cout << "정렬 완료\n";
}
